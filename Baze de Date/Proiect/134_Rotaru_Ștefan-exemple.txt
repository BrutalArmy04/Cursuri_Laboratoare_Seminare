SELECT 
    J.NUME_UTILIZATOR AS NUME,
    NVL((
        SELECT SUM(R.CANTITATE) 
        FROM RES_JOC_JOC RJJ 
        JOIN RESURSA R ON RJJ.ID_RES = R.ID_RESURSA
        WHERE RJJ.NUME_UTILIZATOR = J.NUME_UTILIZATOR
    ), 0) AS TOTAL_RESURSE,
    C.DATA_INREGISTRARE,
    SUBSTR(J.NUME_UTILIZATOR, 1, 3) || '***' AS NUME_ASCUNS,
    DECODE(
        SIGN(NVL((
            SELECT SUM(R.CANTITATE) 
            FROM RES_JOC_JOC RJJ 
            JOIN RESURSA R ON RJJ.ID_RES = R.ID_RESURSA
            WHERE RJJ.NUME_UTILIZATOR = J.NUME_UTILIZATOR
        ), 0) - 13),
        1, 'BOGAT',
        'NORMAL'
    ) AS STATUT_FINANCIAR
FROM 
    JUCATOR J
JOIN 
    CONT C ON J.ID_CONT = C.ID_CONT
WHERE 
    EXISTS (
        SELECT 1 
        FROM ORAS O
        JOIN CLADIRE CL ON O.ID_ORAS = CL.ID_ORAS
        WHERE O.NUME_UTILIZATOR = J.NUME_UTILIZATOR
        AND CL.POATE_FI_AFECTATA = 0
    )
ORDER BY 
    TOTAL_RESURSE DESC;

SELECT 
    C.NUME_CARACTER,
    A.NUME_ABILITATE,
    COUNT(SC.SEL_CAR) AS NR_UTILIZARI,
    SUBSTR(A.DESCRIERE, 1, 30) || '...' AS DESCRIERE_SCURTATA,
    TO_CHAR((
        SELECT MAX(DATA_INREGISTRARE) 
        FROM CONT 
        WHERE ID_CONT IN (
            SELECT ID_CONT 
            FROM JUCATOR 
            WHERE NUME_UTILIZATOR IN (
                SELECT NUME_UTILIZATOR 
                FROM SELECTIE_CARACTER
            )
        )
    ), 'MM-YYYY-DD') AS ULTIMA_UTILIZARE,
    RANK() OVER (ORDER BY COUNT(SC.SEL_CAR) DESC) AS CLASAMENT_POPULARITATE,
    CASE 
        WHEN COUNT(SC.SEL_CAR) > 5 THEN 'Popular'
        WHEN COUNT(SC.SEL_CAR) > 2 THEN 'Mediu'
        ELSE 'Puțin utilizat'
    END AS POPULARITATE
FROM CARACTER C
JOIN ABILITATE A ON C.ID_ABILITATE = A.ID_ABILITATE
LEFT JOIN SELECTIE_CARACTER SC ON C.NUME_CARACTER = SC.NUME_CARACTER
WHERE C.NUME_CARACTER IN (
    SELECT NUME_CARACTER 
    FROM (
        SELECT NUME_CARACTER, COUNT(*) AS CNT 
        FROM SELECTIE_CARACTER 
        GROUP BY NUME_CARACTER
        HAVING COUNT(*) > (
            SELECT AVG(COUNT(*)) 
            FROM SELECTIE_CARACTER 
            GROUP BY NUME_CARACTER
        )
    )
)
GROUP BY C.NUME_CARACTER, A.NUME_ABILITATE, A.DESCRIERE
HAVING COUNT(SC.SEL_CAR) > 0
ORDER BY NR_UTILIZARI DESC;

SELECT 
    DISTINCT J.NUME_UTILIZATOR AS NUME,
    C.NUME_CARACTER AS CARACTER,
    JOC.ID_JOC AS JOC,
    TO_CHAR(CONT.DATA_INREGISTRARE, 'DD-MON-YYYY') AS DATA_INREGISTRARE,
    TRUNC(MONTHS_BETWEEN(SYSDATE, CONT.DATA_INREGISTRARE)/12, 1) AS VECHIME_CONT
FROM 
    JUCATOR J
JOIN 
    CONT ON J.ID_CONT = CONT.ID_CONT
JOIN 
    SELECTIE_CARACTER SC ON J.NUME_UTILIZATOR = SC.NUME_UTILIZATOR
JOIN 
    CARACTER C ON SC.NUME_CARACTER = C.NUME_CARACTER
JOIN 
    TURA T ON SC.ID_TURA = T.ID_TURA
JOIN 
    JOC ON T.ID_JOC = JOC.ID_JOC
WHERE 
    JOC.STATUS = 'IN DESFASURARE'
    AND MONTHS_BETWEEN(SYSDATE, CONT.DATA_INREGISTRARE) > 24  
ORDER BY  
    J.NUME_UTILIZATOR;

WITH 
NR_CLADIRI AS (
    SELECT 
        O.NUME_UTILIZATOR,
        COUNT(C.ID_CLADIRE) AS TOTAL_CLADIRI
    FROM 
        ORAS O
    LEFT JOIN 
        CLADIRE C ON O.ID_ORAS = C.ID_ORAS
    GROUP BY 
        O.NUME_UTILIZATOR
),
CLADIRI_MAXIM AS (
    SELECT 
        MAX(TOTAL_CLADIRI) AS MAXIM
    FROM 
        NR_CLADIRI
)
SELECT 
    J.NUME_UTILIZATOR AS PLAYER_NAME,
    NC.TOTAL_CLADIRI,
    C.EMAIL,
    C.DATA_INREGISTRARE AS REGISTRATION_DATE
FROM 
    JUCATOR J
JOIN 
    CONT C ON J.ID_CONT = C.ID_CONT
JOIN 
    NR_CLADIRI NC ON J.NUME_UTILIZATOR = NC.NUME_UTILIZATOR
CROSS JOIN 
    CLADIRI_MAXIM CM
WHERE 
    NC.TOTAL_CLADIRI = CM.MAXIM
ORDER BY 
    NC.TOTAL_CLADIRI DESC;

WITH ORASEVALOROASE AS (
    SELECT O.NUME_UTILIZATOR, SUM(CT.COST) AS VALOARE_TOTALA
    FROM ORAS O
    JOIN CLADIRE C ON O.ID_ORAS = C.ID_ORAS
    JOIN CARTE CT ON C.ID_CARTE = CT.ID_CARTE
    GROUP BY O.NUME_UTILIZATOR
    HAVING SUM(CT.COST) > (
        SELECT AVG(COST) 
        FROM CARTE
    )
)
SELECT 
    J.NUME_UTILIZATOR,
    OV.VALOARE_TOTALA,
    CASE
        WHEN OV.VALOARE_TOTALA > 10 THEN 'Mare'
        ELSE 'Mic'
    END AS TIP_ORAS
FROM JUCATOR J
JOIN ORASEVALOROASE OV ON J.NUME_UTILIZATOR = OV.NUME_UTILIZATOR
ORDER BY OV.VALOARE_TOTALA DESC;

--outer Join

SELECT
    J.NUME_UTILIZATOR AS IGN,
    C.EMAIL,
    C.DATA_INREGISTRARE AS DATA_INREG,
    RJJ.ID_JOC AS NR_JOC,
    O.ID_ORAS AS CITY_ID,
    CT.NUME AS NUME_CL,
    SC.NUME_CARACTER AS CAR_ALES,
    CA.NUMAR AS NR_CAR,
    AB.NUME_ABILITATE AS NUME_AB,
    T.NUME AS RESURSA,
    R.CANTITATE AS NUMAR
FROM
    JUCATOR J
LEFT OUTER JOIN
    CONT C ON J.ID_CONT = C.ID_CONT
LEFT OUTER JOIN
    ORAS O ON J.NUME_UTILIZATOR = O.NUME_UTILIZATOR
LEFT OUTER JOIN
    CLADIRE CL ON O.ID_ORAS = CL.ID_ORAS
LEFT OUTER JOIN
    CARTE CT ON CL.ID_CARTE = CT.ID_CARTE
LEFT OUTER JOIN
    SELECTIE_CARACTER SC ON J.NUME_UTILIZATOR = SC.NUME_UTILIZATOR
LEFT OUTER JOIN
    CARACTER CA ON SC.NUME_CARACTER = CA.NUME_CARACTER
LEFT OUTER JOIN
    ABILITATE AB ON CA.ID_ABILITATE = AB.ID_ABILITATE
LEFT OUTER JOIN
    RES_JOC_JOC RJJ ON J.NUME_UTILIZATOR = RJJ.NUME_UTILIZATOR
LEFT OUTER JOIN
    RESURSA R ON RJJ.ID_RES = R.ID_RESURSA
LEFT OUTER JOIN
    TIP_RESURSA T ON R.TIP = T.TIP
ORDER BY
    J.NUME_UTILIZATOR,
    O.ID_ORAS,
    CL.ID_CLADIRE,
    SC.SEL_CAR,
    RJJ.ID_JOC,
    R.ID_RESURSA;

SELECT J.NUME_UTILIZATOR
FROM JUCATOR J
WHERE NOT EXISTS (
    SELECT RJJ1.ID_JOC
    FROM RES_JOC_JOC RJJ1
    WHERE RJJ1.NUME_UTILIZATOR = 'ariciul_darnic'
    AND NOT EXISTS (
        SELECT 1
        FROM RES_JOC_JOC RJJ2
        WHERE RJJ2.NUME_UTILIZATOR = J.NUME_UTILIZATOR
        AND RJJ2.ID_JOC = RJJ1.ID_JOC
    )
)
AND J.NUME_UTILIZATOR != 'ariciul_darnic';

-- Analiza top N

SELECT *
FROM (
    SELECT 
        J.NUME_UTILIZATOR,
        SUM(C.COST) AS VALOARE_TOTALA_CLADIRI,
        COUNT(CL.ID_CLADIRE) AS NR_CLADIRI,
        RANK() OVER (ORDER BY SUM(C.COST) DESC) AS CLASAMENT
    FROM JUCATOR J
    JOIN ORAS O ON J.NUME_UTILIZATOR = O.NUME_UTILIZATOR
    JOIN CLADIRE CL ON O.ID_ORAS = CL.ID_ORAS
    JOIN CARTE C ON CL.ID_CARTE = C.ID_CARTE
    GROUP BY J.NUME_UTILIZATOR
)
WHERE CLASAMENT <= 3;
