CREATE TABLE CONT (
    id_cont NUMBER(4) CONSTRAINT pk_cont PRIMARY KEY,
    email VARCHAR2(50) CONSTRAINT uk_cont_email UNIQUE NOT NULL,
    parola VARCHAR2(30) NOT NULL,
    data_inregistrare DATE NOT NULL
);

CREATE TABLE JUCATOR (
    id_jucator NUMBER(4) CONSTRAINT pk_jucator PRIMARY KEY,
    nume_utilizator VARCHAR2(20) CONSTRAINT uk_jucator_nume UNIQUE NOT NULL,
    id_cont NUMBER(4) CONSTRAINT fk_jucator_cont REFERENCES CONT(id_cont) NOT NULL
);

CREATE TABLE JOC (
    id_joc NUMBER(4) CONSTRAINT pk_joc PRIMARY KEY,
    status VARCHAR2(15) NOT NULL CHECK (status IN ('IN DESFASURARE', 'FINALIZAT'))
);

CREATE TABLE ABILITATE (
    id_abilitate NUMBER(3) CONSTRAINT pk_abilitate PRIMARY KEY,
    nume_abilitate VARCHAR2(30) CONSTRAINT uk_abilitate_nume UNIQUE NOT NULL,
    descriere VARCHAR2(200) NOT NULL
);

CREATE TABLE CARACTER (
    id_caracter NUMBER(3) CONSTRAINT pk_caracter PRIMARY KEY,
    nume_caracter VARCHAR2(20) CONSTRAINT uk_caracter_nume UNIQUE NOT NULL,
    id_abilitate NUMBER(3) CONSTRAINT fk_caracter_abilitate REFERENCES ABILITATE(id_abilitate) NOT NULL,
    numar NUMBER(1) NOT NULL CHECK (numar BETWEEN 1 AND 8)
);

CREATE TABLE TIP_RESURSA (
    id_tip_res NUMBER(1) CONSTRAINT pk_tip_resursa PRIMARY KEY CHECK(id_tip_res IN (1,2,3,4,5)),
    nume VARCHAR2(15) CONSTRAINT uk_tip_resursa_nume UNIQUE NOT NULL
);

CREATE TABLE CARTE (
    id_carte NUMBER(3) CONSTRAINT pk_carte PRIMARY KEY,
    tip_carte VARCHAR2(15) NOT NULL CHECK (tip_carte IN ('Negustoreasca', 'Boiereasca', 'Religioasa', 'Militara', 'Unica')),
    cost NUMBER(2) NOT NULL CHECK (cost BETWEEN 1 AND 10),
    nume VARCHAR2(30) CONSTRAINT uk_carte_nume UNIQUE NOT NULL
);

CREATE TABLE TURA (
    id_tura NUMBER(4) CONSTRAINT pk_tura PRIMARY KEY,
    id_joc NUMBER(4) CONSTRAINT fk_tura_joc REFERENCES JOC(id_joc) NOT NULL,
    carac_decart_inviz VARCHAR2(20) NOT NULL
);

CREATE TABLE SELECTIE_CARACTER (
    id_selectie NUMBER(4) CONSTRAINT pk_selectie_caracter PRIMARY KEY,
    id_jucator NUMBER(4) CONSTRAINT fk_selectie_jucator REFERENCES JUCATOR(id_jucator) NOT NULL,
    id_caracter NUMBER(3) CONSTRAINT fk_selectie_caracter REFERENCES CARACTER(id_caracter) NOT NULL,
    id_tura NUMBER(4) CONSTRAINT fk_selectie_tura REFERENCES TURA(id_tura) NOT NULL,
    caract_decart_viz VARCHAR2(20)
);

CREATE TABLE RESURSA (
    id_resursa NUMBER(5) CONSTRAINT pk_resursa PRIMARY KEY,
    id_tip_res NUMBER(1) CONSTRAINT fk_resursa_tip REFERENCES TIP_RESURSA(id_tip_res) NOT NULL,
    cantitate NUMBER(3) NOT NULL CHECK (cantitate > 0)
);

CREATE TABLE CARTE_UNICA (
    id_carte_unica NUMBER(3) CONSTRAINT pk_carte_unica PRIMARY KEY,
    id_abilitate NUMBER(3) CONSTRAINT fk_carte_unica_abilitate REFERENCES ABILITATE(id_abilitate) NOT NULL,
    CONSTRAINT fk_carte_unica_carte FOREIGN KEY (id_carte_unica) REFERENCES CARTE(id_carte)
);

CREATE TABLE ORAS (
    id_oras NUMBER(3) CONSTRAINT pk_oras PRIMARY KEY,
    id_jucator NUMBER(4) CONSTRAINT fk_oras_jucator REFERENCES JUCATOR(id_jucator) NOT NULL
);

CREATE TABLE CLADIRE (
    id_cladire NUMBER(4) CONSTRAINT pk_cladire PRIMARY KEY,
    id_carte NUMBER(3) CONSTRAINT fk_cladire_carte REFERENCES CARTE(id_carte) NOT NULL,
    id_oras NUMBER(3) CONSTRAINT fk_cladire_oras REFERENCES ORAS(id_oras) NOT NULL,
    poate_fi_afectata NUMBER(1) NOT NULL CHECK (poate_fi_afectata IN (0, 1))
);

CREATE TABLE FOLOSESTE (
    id_folosire NUMBER(4) CONSTRAINT pk_foloseste PRIMARY KEY,
    id_resursa NUMBER(5) CONSTRAINT fk_foloseste_resursa REFERENCES RESURSA(id_resursa) NOT NULL,
    id_cladire NUMBER(4) CONSTRAINT fk_foloseste_cladire REFERENCES CLADIRE(id_cladire) NOT NULL,
    cantitate_folosita NUMBER(2) NOT NULL CHECK (cantitate_folosita BETWEEN 1 AND 10),
    CONSTRAINT uk_foloseste_res_clad UNIQUE (id_resursa, id_cladire)
);

CREATE TABLE RES_JOC_JUC (
    id_relatie NUMBER(4) CONSTRAINT pk_res_joc_juc PRIMARY KEY,
    id_jucator NUMBER(4) CONSTRAINT fk_res_joc_juc_jucator REFERENCES JUCATOR(id_jucator) NOT NULL,
    id_joc NUMBER(4) CONSTRAINT fk_res_joc_juc_joc REFERENCES JOC(id_joc) NOT NULL,
    id_resursa NUMBER(5) CONSTRAINT fk_res_joc_juc_res REFERENCES RESURSA(id_resursa) NOT NULL,
    CONSTRAINT uk_res_joc_juc_res UNIQUE (id_jucator, id_joc, id_resursa)
);

CREATE OR REPLACE PROCEDURE raport_avere_jucator (
    p_nume_utilizator IN JUCATOR.nume_utilizator%TYPE,
    p_id_joc IN JOC.id_joc%TYPE
) IS

    TYPE t_resurse_map IS TABLE OF NUMBER INDEX BY VARCHAR2(50);
    v_harta_resurse t_resurse_map;
    
    TYPE t_lista_cladiri IS TABLE OF VARCHAR2(50);
    v_cladiri_construite t_lista_cladiri := t_lista_cladiri();
    
    TYPE t_top_caractere IS VARRAY(3) OF VARCHAR2(50);
    v_istoric_caractere t_top_caractere := t_top_caractere();
    
    v_id_jucator JUCATOR.id_jucator%TYPE;
    v_idx VARCHAR2(50); 
    
    CURSOR c_resurse IS
        SELECT tr.nume, r.cantitate
        FROM RES_JOC_JUC rjj
        JOIN RESURSA r ON rjj.id_resursa = r.id_resursa
        JOIN TIP_RESURSA tr ON r.id_tip_res = tr.id_tip_res
        WHERE rjj.id_jucator = v_id_jucator AND rjj.id_joc = p_id_joc;

BEGIN
    SELECT id_jucator INTO v_id_jucator 
    FROM JUCATOR WHERE nume_utilizator = p_nume_utilizator;

    DBMS_OUTPUT.PUT_LINE('=== RAPORT PENTRU JUCATORUL: ' || p_nume_utilizator || ' ===');

    FOR rec IN c_resurse LOOP
        IF v_harta_resurse.EXISTS(rec.nume) THEN
            v_harta_resurse(rec.nume) := v_harta_resurse(rec.nume) + rec.cantitate;
        ELSE
             v_harta_resurse(rec.nume) := rec.cantitate;
        END IF;
    END LOOP;
    
    DBMS_OUTPUT.PUT_LINE('--- Resurse Detinute ---');
    v_idx := v_harta_resurse.FIRST;
    WHILE v_idx IS NOT NULL LOOP
         DBMS_OUTPUT.PUT_LINE('Resursa ' || v_idx || ': ' || v_harta_resurse(v_idx));
         v_idx := v_harta_resurse.NEXT(v_idx);
    END LOOP;
    
    IF v_harta_resurse.COUNT = 0 THEN
        DBMS_OUTPUT.PUT_LINE('Nu detine resurse in acest joc.');
    END IF;

    SELECT c.nume
    BULK COLLECT INTO v_cladiri_construite
    FROM CLADIRE cl
    JOIN CARTE c ON cl.id_carte = c.id_carte
    JOIN ORAS o ON cl.id_oras = o.id_oras
    WHERE o.id_jucator = v_id_jucator;
    
    DBMS_OUTPUT.PUT_LINE('--- Cladiri Construite (' || v_cladiri_construite.COUNT || ') ---');
    IF v_cladiri_construite.COUNT > 0 THEN
        FOR i IN v_cladiri_construite.FIRST .. v_cladiri_construite.LAST LOOP
            DBMS_OUTPUT.PUT_LINE('Cladire: ' || v_cladiri_construite(i));
        END LOOP;
    ELSE
        DBMS_OUTPUT.PUT_LINE('Nicio cladire construita.');
    END IF;

    FOR rec IN (SELECT DISTINCT c.nume_caracter 
                FROM SELECTIE_CARACTER sc
                JOIN CARACTER c ON sc.id_caracter = c.id_caracter
                JOIN TURA t ON sc.id_tura = t.id_tura  
                WHERE sc.id_jucator = v_id_jucator AND t.id_joc = p_id_joc 
                ORDER BY c.nume_caracter FETCH FIRST 2 ROWS ONLY) 
    LOOP
        v_istoric_caractere.EXTEND;
        v_istoric_caractere(v_istoric_caractere.LAST) := rec.nume_caracter;
    END LOOP;
    
    DBMS_OUTPUT.PUT_LINE('--- Caractere Jucate (Top 2 alfabetic) ---');
    IF v_istoric_caractere.COUNT > 0 THEN
        FOR i IN 1 .. v_istoric_caractere.COUNT LOOP
             DBMS_OUTPUT.PUT_LINE('Caracter: ' || v_istoric_caractere(i));
        END LOOP;
    ELSE
         DBMS_OUTPUT.PUT_LINE('Nu a jucat nicio tura in acest joc.');
    END IF;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Jucatorul nu a fost gasit sau nu exista date.');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Eroare neasteptata: ' || SQLERRM);
END raport_avere_jucator;
/

END;

CREATE OR REPLACE PROCEDURE analiza_jocuri_finalizate IS
    CURSOR c_jocuri IS
        SELECT id_joc, status 
        FROM JOC 
        WHERE status = 'FINALIZAT';
        
    CURSOR c_detalii_jucatori (p_id_joc NUMBER) IS
        SELECT j.nume_utilizator, c.nume_caracter
        FROM SELECTIE_CARACTER sc
        JOIN JUCATOR j ON sc.id_jucator = j.id_jucator
        JOIN CARACTER c ON sc.id_caracter = c.id_caracter
        JOIN TURA t ON sc.id_tura = t.id_tura
        WHERE t.id_joc = p_id_joc
        AND t.id_tura = (SELECT MIN(id_tura) FROM TURA WHERE id_joc = p_id_joc);

    v_id_joc_curent JOC.id_joc%TYPE;
    v_status JOC.status%TYPE;
    
    v_nume_user JUCATOR.nume_utilizator%TYPE;
    v_nume_caracter CARACTER.nume_caracter%TYPE;

BEGIN
    OPEN c_jocuri;
    LOOP
        FETCH c_jocuri INTO v_id_joc_curent, v_status;
        EXIT WHEN c_jocuri%NOTFOUND;
        
        DBMS_OUTPUT.PUT_LINE('-------------------------------------------');
        DBMS_OUTPUT.PUT_LINE('Analiza Jocului #' || v_id_joc_curent || ' (Status: ' || v_status || ')');
        DBMS_OUTPUT.PUT_LINE('Jucatori si caracterele de start:');
        
        OPEN c_detalii_jucatori(v_id_joc_curent);
        LOOP
            FETCH c_detalii_jucatori INTO v_nume_user, v_nume_caracter;
            EXIT WHEN c_detalii_jucatori%NOTFOUND;
            
            DBMS_OUTPUT.PUT_LINE(' -> Jucator: ' || v_nume_user || ' | Caracter Start: ' || v_nume_caracter);
        END LOOP;
        
        IF c_detalii_jucatori%ROWCOUNT = 0 THEN
            DBMS_OUTPUT.PUT_LINE(' -> Nu exista date despre prima tura pentru acest joc.');
        END IF;
        
        CLOSE c_detalii_jucatori; 
        
    END LOOP;
    CLOSE c_jocuri; 

EXCEPTION
    WHEN OTHERS THEN
        IF c_jocuri%ISOPEN THEN CLOSE c_jocuri; END IF;
        IF c_detalii_jucatori%ISOPEN THEN CLOSE c_detalii_jucatori; END IF;
        RAISE_APPLICATION_ERROR(-20003, 'Eroare in analiza jocurilor: ' || SQLERRM);
END analiza_jocuri_finalizate;
/

BEGIN
    analiza_jocuri_finalizate;
END;
/

CREATE OR REPLACE FUNCTION get_cladire_jucator_carte (
    p_nume_utilizator IN JUCATOR.nume_utilizator%TYPE,
    p_id_carte        IN CARTE.id_carte%TYPE
) 
RETURN NUMBER 
IS
    v_id_cladire CLADIRE.id_cladire%TYPE;
BEGIN
    SELECT c.id_cladire
    INTO v_id_cladire
    FROM JUCATOR j
    JOIN ORAS o ON j.id_jucator = o.id_jucator
    JOIN CLADIRE c ON o.id_oras = c.id_oras
    WHERE j.nume_utilizator = p_nume_utilizator
      AND c.id_carte = p_id_carte;

    RETURN v_id_cladire;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Exceptie: Jucatorul ' || p_nume_utilizator || 
                             ' nu are construita cartea cu ID ' || p_id_carte);
        RETURN NULL; 
        
    WHEN TOO_MANY_ROWS THEN
        DBMS_OUTPUT.PUT_LINE('Exceptie: Jucatorul ' || p_nume_utilizator || 
                             ' are mai multe cladiri de tipul cartii ' || p_id_carte);
        RETURN NULL;
        
    WHEN OTHERS THEN
        RAISE_APPLICATION_ERROR(-20002, 'Alta eroare: ' || SQLERRM);
END get_cladire_jucator_carte;
/

CREATE OR REPLACE PROCEDURE detalii_consum_resurse (
    p_nume_utilizator IN JUCATOR.nume_utilizator%TYPE,
    p_tip_carte       IN CARTE.tip_carte%TYPE
) 
IS
    v_count_user    NUMBER;
    v_found_data    BOOLEAN := FALSE;
BEGIN
    SELECT COUNT(*) INTO v_count_user 
    FROM JUCATOR 
    WHERE nume_utilizator = p_nume_utilizator;
    
    IF v_count_user = 0 THEN
        RAISE_APPLICATION_ERROR(-20001, 'Jucatorul "' || p_nume_utilizator || '" nu exista in baza de date.');
    END IF;

    IF p_tip_carte NOT IN ('Negustoreasca', 'Boiereasca', 'Religioasa', 'Militara', 'Unica') THEN
        RAISE_APPLICATION_ERROR(-20002, 'Tipul "' || p_tip_carte || '" nu este valid. Alegeti un tip corect.');
    END IF;

    FOR rec IN (
        SELECT c.nume AS nume_cladire, 
               tr.nume AS nume_resursa, 
               f.cantitate_folosita
        FROM JUCATOR j
        JOIN ORAS o ON j.id_jucator = o.id_jucator
        JOIN CLADIRE cl ON o.id_oras = cl.id_oras
        JOIN CARTE c ON cl.id_carte = c.id_carte
        JOIN FOLOSESTE f ON cl.id_cladire = f.id_cladire
        JOIN RESURSA r ON f.id_resursa = r.id_resursa
        JOIN TIP_RESURSA tr ON r.id_tip_res = tr.id_tip_res
        WHERE j.nume_utilizator = p_nume_utilizator
          AND c.tip_carte = p_tip_carte
    ) LOOP
        v_found_data := TRUE;
        DBMS_OUTPUT.PUT_LINE('Jucatorul ' || p_nume_utilizator || 
                             ' a consumat ' || rec.cantitate_folosita || 
                             ' ' || rec.nume_resursa || 
                             ' pentru a construi ' || rec.nume_cladire || '.');
    END LOOP;

    IF NOT v_found_data THEN
        DBMS_OUTPUT.PUT_LINE('Info: Jucatorul ' || p_nume_utilizator || 
                             ' nu are constructii de tipul ' || p_tip_carte || 
                             ' sau nu au fost inregistrate costuri.');
    END IF;

EXCEPTION
    WHEN OTHERS THEN
        IF SQLCODE BETWEEN -20999 AND -20000 THEN
            RAISE;
        ELSE
            RAISE_APPLICATION_ERROR(-20003, 'Eroare interna SQL: ' || SQLERRM);
        END IF;
END detalii_consum_resurse;
/

DECLARE
    e_user_inexistent EXCEPTION;
    PRAGMA EXCEPTION_INIT(e_user_inexistent, -20001);
    
    e_tip_invalid     EXCEPTION;
    PRAGMA EXCEPTION_INIT(e_tip_invalid, -20002);
BEGIN
    DBMS_OUTPUT.PUT_LINE('------------------------------------------------');
    DBMS_OUTPUT.PUT_LINE('Caz 1: Apel cu jucator inexistent');
    BEGIN
        detalii_consum_resurse('fantoma', 'Religioasa');
    EXCEPTION
        WHEN e_user_inexistent THEN
            DBMS_OUTPUT.PUT_LINE(' Prins eroare -20001: ' || SQLERRM);
    END;

    DBMS_OUTPUT.PUT_LINE('------------------------------------------------');
    DBMS_OUTPUT.PUT_LINE('Caz 2: Apel cu tip de carte gresit');
    BEGIN
        detalii_consum_resurse('lupul_alb', 'Magica');
    EXCEPTION
        WHEN e_tip_invalid THEN
            DBMS_OUTPUT.PUT_LINE(' Prins eroare -20002: ' || SQLERRM);
    END;

    DBMS_OUTPUT.PUT_LINE('------------------------------------------------');
    DBMS_OUTPUT.PUT_LINE('Caz 3: Apel corect (Detalii Resurse)');
    BEGIN
        detalii_consum_resurse('lupul_alb', 'Religioasa');
    EXCEPTION
        WHEN OTHERS THEN
            DBMS_OUTPUT.PUT_LINE(' -> Eroare neasteptata: ' || SQLERRM);
    END;
    DBMS_OUTPUT.PUT_LINE('------------------------------------------------');
END;

CREATE OR REPLACE PACKAGE pkg_administrare_joc IS
    TYPE rec_detalii_jucator IS RECORD (
        nume_utilizator VARCHAR2(50),
        punctaj_cladiri NUMBER,
        resurse_ramase  NUMBER
    );
    
    TYPE t_clasament IS TABLE OF rec_detalii_jucator INDEX BY PLS_INTEGER;

    FUNCTION f_calculeaza_punctaj(p_id_jucator IN JUCATOR.id_jucator%TYPE) 
        RETURN NUMBER;
        
    FUNCTION f_get_total_resurse(p_id_jucator IN JUCATOR.id_jucator%TYPE, p_id_joc IN JOC.id_joc%TYPE) 
        RETURN NUMBER;

    PROCEDURE p_afiseaza_clasament(p_id_joc IN JOC.id_joc%TYPE);
    
    PROCEDURE p_finalizare_joc(p_id_joc IN JOC.id_joc%TYPE);

END pkg_administrare_joc;
/

CREATE OR REPLACE PACKAGE BODY pkg_administrare_joc IS

    FUNCTION f_calculeaza_punctaj(p_id_jucator IN JUCATOR.id_jucator%TYPE) 
    RETURN NUMBER IS
        v_total_puncte NUMBER := 0;
    BEGIN
        SELECT NVL(SUM(c.cost), 0)
        INTO v_total_puncte
        FROM CLADIRE cl
        JOIN CARTE c ON cl.id_carte = c.id_carte
        JOIN ORAS o ON cl.id_oras = o.id_oras
        WHERE o.id_jucator = p_id_jucator;
        
        RETURN v_total_puncte;
    EXCEPTION
        WHEN OTHERS THEN
            RETURN 0;
    END f_calculeaza_punctaj;


    FUNCTION f_get_total_resurse(p_id_jucator IN JUCATOR.id_jucator%TYPE, p_id_joc IN JOC.id_joc%TYPE) 
    RETURN NUMBER IS
        v_total_res NUMBER := 0;
    BEGIN
        SELECT NVL(SUM(r.cantitate), 0)
        INTO v_total_res
        FROM RES_JOC_JUC rjj
        JOIN RESURSA r ON rjj.id_resursa = r.id_resursa
        WHERE rjj.id_jucator = p_id_jucator AND rjj.id_joc = p_id_joc;
        
        RETURN v_total_res;
    EXCEPTION
        WHEN OTHERS THEN
            RETURN 0;
    END f_get_total_resurse;

    PROCEDURE p_afiseaza_clasament(p_id_joc IN JOC.id_joc%TYPE) IS
        v_clasament t_clasament; 
        v_idx PLS_INTEGER := 0;
    BEGIN
        DBMS_OUTPUT.PUT_LINE('--- Clasament JOC ' || p_id_joc || ' ---');
        
        FOR r IN (SELECT DISTINCT id_jucator 
                  FROM RES_JOC_JUC 
                  WHERE id_joc = p_id_joc) 
        LOOP
            v_idx := v_idx + 1;

            SELECT nume_utilizator INTO v_clasament(v_idx).nume_utilizator
            FROM JUCATOR WHERE id_jucator = r.id_jucator;
            
            v_clasament(v_idx).punctaj_cladiri := f_calculeaza_punctaj(r.id_jucator);
            v_clasament(v_idx).resurse_ramase  := f_get_total_resurse(r.id_jucator, p_id_joc);
            
            DBMS_OUTPUT.PUT_LINE(v_idx || '. ' || v_clasament(v_idx).nume_utilizator || 
                                 ' | Puncte Cladiri: ' || v_clasament(v_idx).punctaj_cladiri ||
                                 ' | Resurse: ' || v_clasament(v_idx).resurse_ramase);
        END LOOP;
        
        IF v_idx = 0 THEN
            DBMS_OUTPUT.PUT_LINE('Nu exista jucatori inregistrati pentru acest joc.');
        END IF;
    END p_afiseaza_clasament;

    PROCEDURE p_finalizare_joc(p_id_joc IN JOC.id_joc%TYPE) IS
        v_status_actual JOC.status%TYPE;
    BEGIN
        SELECT status INTO v_status_actual FROM JOC WHERE id_joc = p_id_joc;
        
        IF v_status_actual = 'FINALIZAT' THEN
            DBMS_OUTPUT.PUT_LINE('Atentie: Jocul ' || p_id_joc || ' este deja finalizat.');
            p_afiseaza_clasament(p_id_joc);
        ELSE
            p_afiseaza_clasament(p_id_joc);
            
            UPDATE JOC SET status = 'FINALIZAT' WHERE id_joc = p_id_joc;
            DBMS_OUTPUT.PUT_LINE('Succes: Jocul ' || p_id_joc || ' a fost marcat ca FINALIZAT.');
        END IF;
        
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RAISE_APPLICATION_ERROR(-20010, 'Jocul cu ID ' || p_id_joc || ' nu exista.');
        WHEN OTHERS THEN
            RAISE_APPLICATION_ERROR(-20011, 'Eroare la finalizare: ' || SQLERRM);
    END p_finalizare_joc;

END pkg_administrare_joc;
/

    pkg_administrare_joc.p_finalizare_joc(1);
    
EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Eroare prinsa in blocul de test: ' || SQLERRM);
END;
